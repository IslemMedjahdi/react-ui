# Alert

This alert component can be used to highlight a message to the user.
It supports different types of tones.

export const Template = ({ children, ...args }) => (
  <Alert {...args}>{children}</Alert>
)

<Canvas>
  <Story
    name="Default"
    args={{ children: 'Critical Message.', tone: Tone.critical }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
