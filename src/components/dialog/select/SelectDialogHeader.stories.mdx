# SelectDialogHeader

This component is mainly used in the [SelectDialog](/docs/components-header-selectdialog--default-story) to provide there an easy convenient solution of adding a search.

It leverages on the component [HeaderSearch](/docs/components-header-headersearch--default-story) which provides the search logic.

export const Template = ({
  buttonText,
  buttonTone,
  title,
  iconLabel,
  ...args
}) => {
  const [toggle, setToggle] = useState(false)
  const [text, setText] = useState('')
  const clearSearch = () => setText('')
  return (
    <NavigationProvider>
      {toggle ? (
        <SelectDialogHeader
          {...args}
          title={title}
          iconLabel={iconLabel}
          search={text}
          actions={{ search: setText, clear: clearSearch }}
          onDismiss={() => setToggle(!toggle)}
        />
      ) : (
        <Button tone={buttonTone} onClick={() => setToggle(!toggle)}>
          {buttonText}
        </Button>
      )}
    </NavigationProvider>
  )
}
