# SectionSearch

This component defines the entire search part that can be placed inside the `SectionHeader`. It wraps a `SectionTitle`
and a `SectionAction` for not searching and an input with another `SectionAction` for searching.

export const Template = ({ ...args }) => {
  const [search, setSearch] = useState('')
  return (
    <div className="flex items-center">
      <SectionSearch
        {...args}
        search={search}
        actions={{ search: setSearch, clear: () => setSearch('') }}
      />
    </div>
  )
}
