# SectionListItemWithButton

This component adds an HTML-button element and applies basic styling.

export const Template = (args) => {
  const array = [1, 2, 3]
  return (
    <SectionContentList>
      {array.map((elem) => (
        <SectionListItemWithButtonComponent
          {...args}
          key={elem}
          onClick={() => {
            alert(`Button ${elem} was clicked!`)
          }}
        >
          Button {elem}
        </SectionListItemWithButtonComponent>
      ))}
    </SectionContentList>
  )
}
