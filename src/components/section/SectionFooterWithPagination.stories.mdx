# SectionFooterWithPagination

This component can conveniently be used to add a pagination with a section footer. This component leverages on [SectionFooter](/docs/components-section-sectionfooter--section-footer) and [PaginationRouter](/docs/components-pagination-paginationrouter--default-story).

export const Template = () => {
  const currentParams = new URLSearchParams(window.parent.location.search)
  const currentPage = parseInt(currentParams.get('page'), 10) || 1
  const currentSize = parseInt(currentParams.get('size'), 10) || 10
  return (
    <SectionFooterWithPagination
      linkProps={({ pageIndex, size }) => {
        const params = new URLSearchParams(window.parent.location.search)
        params.set('page', String(pageIndex))
        params.set('size', String(size))
        return {
          href: window.parent.location.pathname + '?' + params.toString(),
        }
      }}
      size={currentSize}
      page={currentPage}
      total={60}
    />
  )
}
