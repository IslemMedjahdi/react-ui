# SectionPaginationInMemory

This component is used for an in memory pagination.

It leverages on the components [SectionPaginationContainer](/docs/components-section-SectionPaginationContainer--default-story), [SectionPaginationNextContent](/docs/components-section-SectionPaginationNextContent--default-story), [SectionPaginationPreviousContent](/docs/components-section-SectionPaginationPreviousContent--default-story) , [SectionPaginationPagesList](/docs/components-section-SectionPaginationPagesList--default-story), and [SectionPaginationPagesListItem](/docs/components-section-SectionPaginationPagesListItem--default-story).

export const Template = ({ children, ...args }) => {
  const [page, setPage] = useState(1)
  return (
    <SectionPaginationInMemory
      {...args}
      size={5}
      page={page}
      total={60}
      nextLabel="Next page"
      previousLabel="Previous page"
      paginationActions={{ setPage: setPage }}
    >
      {children}
    </SectionPaginationInMemory>
  )
}

<Canvas>
