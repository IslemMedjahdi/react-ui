# HeaderMain

This component can be used to creat swiftly a header.

**_Important_**: this component needs to be wrapped in a `IntlProvider` and `MenuProvider`. The detailed description of them can be found in the section of [HeaderArea](/docs/components-header-headerarea--default-story).

export const Template = ({ ...args }) => (
  <IntlProvider
    locale="en"
    messages={{
      'app.nav.menu': 'Menu',
    }}
  >
    <MenuProvider>
      <HeaderMain {...args} title="Boots Shop" />
    </MenuProvider>
  </IntlProvider>
)

  <Story name="Default" args={{ children: 'Boots Shop' }}>
    {Template.bind({})}
  </Story>
</Canvas>
