# HeaderMainWithSearch

This component is similar to [HeaderMain](/docs/components-header-headermain--default-story), but with the difference that a search button can be found on the right side.

It leverages on the component [HeaderSearch](/docs/components-header-headersearch--default-story).

export const Template = ({ ...args }) => {
  const [text, setText] = useState('')
  const clearSearch = () => setText('')
  return (
    <MenuProvider>
      <HeaderMainWithSearch
        {...args}
        search={text}
        searchActions={{ search: setText, clear: clearSearch }}
        label="open search"
        title="Header Title"
      />
    </MenuProvider>
  )
}
