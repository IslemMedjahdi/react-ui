# Form

This component is a wrapper around [Formik's form](https://formik.org/docs/api/form) component.
It also applies a gap between its children.

export const Template = ({ children, ...args }) => (
  <Formik
    onSubmit={(values) => {
      console.log(values)
    }}
    initialValues={{ email: '' }}
  >
    <Form {...args}>{children}</Form>
  </Formik>
)
