# MenuItem

This component is used to add an item to the [Menu](/docs/components-menu-menu--default-story).

It leverages on the npm package [@reach/menu-button](https://www.npmjs.com/package/@reach/menu-button).

export const Template = () => (
  <Menu menuLabel="Menu" text="Menu" className="lg:bg-primary">
    <MenuItem onClick={() => alert(`Menu item 1 was pressed`)} key={1}>
      {`Menu item 1`}
    </MenuItem>
  </Menu>
)
