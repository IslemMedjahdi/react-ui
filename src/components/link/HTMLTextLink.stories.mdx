# HtmlTextLink

This component is used to create a link to an external page.

It leverages on the HTML anchor (`a`).

export const Template = ({ children, href, ...args }) => (
  <HtmlTextLink {...args} href={href}>
    {children}
  </HtmlTextLink>
)
